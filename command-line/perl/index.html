to run

.pl files are run by the perl interpreter

$ perl script.pl
Hello, Perl!

script.pl

#!/usr/bin/perl
print "Hello, Perl!\n";

more perl

#!/usr/bin/perl
# print the ASCII value of each character in a string
$str = "Hello, Perl!";
for $i (0..length($str)-1) {
    print substr($str, $i, 1) . " - " . ord(substr($str, $i, 1)) . "\n";
}
